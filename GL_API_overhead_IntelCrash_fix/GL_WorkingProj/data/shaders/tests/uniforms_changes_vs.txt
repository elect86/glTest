#version 150 core

in vec3 s_pos;
in vec3 s_normal;
in vec2 s_uv;
in vec4 s_attribute_3; //color

uniform mat4 ModelViewProjectionMatrix;
uniform vec4 color0;
uniform vec4 color1;
uniform vec4 color2;
uniform vec4 color3;
uniform vec4 color4;
uniform vec4 color5;
uniform vec4 color6;
uniform vec4 color7;
uniform vec4 color8;
uniform vec4 color9;

out vec3 instance_color;

void main()
{
	//just use/read data from uniforms
	instance_color = fract(color0.xyz + color1.xyz + color2.xyz + color3.xyz + color4.xyz + color5.xyz + color6.xyz + color7.xyz + color8.xyz + color9.xyz);
	gl_Position = ModelViewProjectionMatrix * vec4(s_pos, 1.0);
}
